impl Strategy for NewFunctionsTest {
    fn on_bar(&mut self, bars: &[Bar], orders: &mut Vec<Order>) {
        let n = bars.len();
        if n < 2 { return; }
        let bar = &bars[n - 1];
        let prev = &bars[n - 2];
        let close = bar.close;
        let high = bar.high;
        let low = bar.low;
        let open = bar.open;
        let volume = bar.volume;


        // =================================================================
        // SECTION 1 — Moving Averages
        // =================================================================

        // PL: TriAverage(Close, 14)
        //   → manual: triangular MA — SMA of SMA with halved length,
        //   half_len = ceil((14 + 1) / 2) = 8 (two chained SMAs, NOT three).
        let tri_inner = self.tri_sma1.next(close);
        let tri_avg = self.tri_sma2.next(tri_inner);

        // PL: SmoothedAverage(Close, 14)
        //   → manual: Wilder smoothing: prev*(N-1)/N + val/N
        //   Equivalent to Wilder's EMA with period N.
        let sm_n = 14.0_f64;
        if self.bar_count == 0 {
            self.smoothed_avg = close;
        } else {
            self.smoothed_avg = self.smoothed_avg * (sm_n - 1.0) / sm_n + close / sm_n;
        }
        let smoothed_avg = self.smoothed_avg;


        // =================================================================
        // SECTION 2 — Stochastic Helpers (9 functions)
        // =================================================================

        // PL: FastK(14)
        //   → manual: 100 * (close - lowest_low_14) / (highest_high_14 - lowest_low_14)
        let hh = self.stoch_hi14.next(high);
        let ll = self.stoch_lo14.next(low);
        let fast_k = if hh > ll {
            100.0 * (close - ll) / (hh - ll)
        } else {
            0.0
        };

        // PL: FastD(14)
        //   → manual: SMA(3) of FastK
        let fast_d = self.fast_k_sma.next(fast_k);

        // PL: SlowK(14)
        //   → manual: SMA(3) of FastK (same as FastD)
        let slow_k = fast_d;

        // PL: SlowD(14)
        //   → manual: SMA(3) of SlowK
        let slow_d = self.slow_k_sma.next(slow_k);

        // PL: FastKCustom(H, L, C, 14)
        //   → same FastK formula but with custom high/low/close series.
        //   When using the primary bar's H/L/C, identical to FastK.
        //   For a secondary data feed, pass bars2.high / bars2.low / bars2.close.
        let fast_k_custom = fast_k; // same formula, custom inputs would differ

        // PL: FastDCustom(H, L, C, 14)
        //   → SMA(3) of FastKCustom
        let fast_d_custom = fast_d; // SMA(3) of custom FastK

        // PL: SlowKCustom(H, L, C, 14)
        //   → SMA(3) of FastKCustom (same as FastDCustom)
        let slow_k_custom = fast_d_custom;

        // PL: SlowDCustom(H, L, C, 14)
        //   → SMA(3) of SlowKCustom
        let slow_d_custom = slow_d; // SMA(3) of above

        // PL: StochasticExp(H, L, C, 14, 3, 3, ...)
        //   → manual: EMA smoothing of FastKCustom instead of SMA
        //   ExpSlowK = EMA(3) of FastK; ExpSlowD = EMA(3) of ExpSlowK
        let stoch_exp_k = self.stoch_exp_ema_k.next(fast_k);
        let stoch_exp_d = self.stoch_exp_ema_d.next(stoch_exp_k);


        // =================================================================
        // SECTION 3 — DM/ADX Variants (6 functions)
        // =================================================================

        let dm_n = 14.0_f64;

        // --- Shared DI+/DI- computation (Wilder smoothing) ---
        let up_move = high - prev.high;
        let dn_move = prev.low - low;
        let pdm = if up_move > dn_move && up_move > 0.0 { up_move } else { 0.0 };
        let mdm = if dn_move > up_move && dn_move > 0.0 { dn_move } else { 0.0 };
        let tr = (high - low)
            .max((high - prev.close).abs())
            .max((low - prev.close).abs());

        if self.bar_count < 14 {
            self.smoothed_pdm += pdm;
            self.smoothed_mdm += mdm;
            self.smoothed_tr += tr;
        } else {
            self.smoothed_pdm = self.smoothed_pdm - self.smoothed_pdm / dm_n + pdm;
            self.smoothed_mdm = self.smoothed_mdm - self.smoothed_mdm / dm_n + mdm;
            self.smoothed_tr = self.smoothed_tr - self.smoothed_tr / dm_n + tr;
        }

        let di_plus = if self.smoothed_tr > 0.0 {
            100.0 * self.smoothed_pdm / self.smoothed_tr
        } else { 0.0 };
        let di_minus = if self.smoothed_tr > 0.0 {
            100.0 * self.smoothed_mdm / self.smoothed_tr
        } else { 0.0 };

        let di_sum = di_plus + di_minus;
        let dx = if di_sum > 0.0 {
            100.0 * (di_plus - di_minus).abs() / di_sum
        } else { 0.0 };

        // PL: DMI(14)
        //   → manual: directional movement index = DX value before ADX smoothing
        let dmi_val = dx;

        // PL: DMIPlusCustom(H, L, C, 14)
        //   → manual: +DI with custom price inputs (same formula, custom feeds)
        let dmi_plus_custom = di_plus;

        // PL: DMIMinusCustom(H, L, C, 14)
        //   → manual: -DI with custom price inputs
        let dmi_minus_custom = di_minus;

        // PL: ADX(14) — Wilder-smoothed DX (base computation)
        if self.bar_count >= 14 {
            if self.dx_count < 14 {
                self.dx_sum += dx;
                self.dx_count += 1;
                if self.dx_count == 14 {
                    self.adx_val = self.dx_sum / dm_n;
                }
            } else {
                self.adx_val = (self.adx_val * (dm_n - 1.0) + dx) / dm_n;
            }
        }

        // PL: ADXR(14)
        //   → manual: (ADX + ADX[14]) / 2
        //   Requires storing ADX history; simplified here using lookback on bars.
        let adxr = if n > 14 {
            // In production, store adx_history[14] ring buffer
            // adxr = (self.adx_val + self.adx_history[14_bars_ago]) / 2.0
            self.adx_val // placeholder — full impl needs ADX history ring buffer
        } else {
            self.adx_val
        };

        // PL: ADXCustom(H, L, C, 14)
        //   → manual: same ADX computation but with custom price inputs
        //   When using bars2 high/low/close, substitute those values into the
        //   DI+/DI-/DX/ADX pipeline above.
        let adx_custom = self.adx_val; // same pipeline, different price feed

        // PL: ParabolicCustom(0.02, 0.2)
        //   → manual: SAR state machine with configurable af_step and af_limit
        //   ParabolicCustom lets you set af_step (default 0.02) and af_limit (default 0.2)
        //   separately from the standard Parabolic(af_step) which uses 0.2 as max.
        if self.bar_count == 0 {
            self.sar = low;
            self.sar_ep = high;
        } else {
            self.sar += self.sar_af * (self.sar_ep - self.sar);
            if self.sar_rising {
                if high > self.sar_ep {
                    self.sar_ep = high;
                    self.sar_af = (self.sar_af + 0.02).min(0.2);
                }
                if low < self.sar {
                    self.sar_rising = false;
                    self.sar = self.sar_ep;
                    self.sar_ep = low;
                    self.sar_af = 0.02;
                }
            } else {
                if low < self.sar_ep {
                    self.sar_ep = low;
                    self.sar_af = (self.sar_af + 0.02).min(0.2);
                }
                if high > self.sar {
                    self.sar_rising = true;
                    self.sar = self.sar_ep;
                    self.sar_ep = high;
                    self.sar_af = 0.02;
                }
            }
        }
        let parabolic_custom = self.sar;


        // =================================================================
        // SECTION 4 — Additional Oscillators (8 functions)
        // =================================================================

        // PL: TRIX(Close, 15)
        //   → manual: 1-period ROC of triple EMA
        //   trix = 100 * (tripleEMA - prevTripleEMA) / prevTripleEMA
        let e1 = self.trix_ema1.next(close);
        let e2 = self.trix_ema2.next(e1);
        let e3 = self.trix_ema3.next(e2);
        let trix_val = if self.prev_trix_triple != 0.0 {
            100.0 * (e3 - self.prev_trix_triple) / self.prev_trix_triple
        } else {
            0.0
        };
        self.prev_trix_triple = e3;

        // PL: MassIndex(9, 25)
        //   → manual: sum over 25 bars of EMA(H-L, 9) / EMA(EMA(H-L, 9), 9)
        let hl_range = high - low;
        let mass_ema1_val = self.mass_ema1.next(hl_range);
        let mass_ema2_val = self.mass_ema2.next(mass_ema1_val);
        let mass_ratio = if mass_ema2_val != 0.0 {
            mass_ema1_val / mass_ema2_val
        } else {
            1.0
        };
        // Full MassIndex sums mass_ratio over 25 bars (needs ring buffer in production)
        let _mass_index = mass_ratio * 25.0; // simplified single-bar approximation

        // PL: EaseOfMovement
        //   → manual: ((H+L)/2 - (prevH+prevL)/2) / ((Volume/1e6) / (H-L))
        let midpoint_move = (high + low) / 2.0 - (prev.high + prev.low) / 2.0;
        let box_ratio = if hl_range > 0.0 {
            (volume / 1_000_000.0) / hl_range
        } else {
            1.0
        };
        let ease_of_movement = if box_ratio != 0.0 {
            midpoint_move / box_ratio
        } else {
            0.0
        };

        // PL: SwingIndex
        //   → manual: Wilder's Swing Index formula
        //   SI = 50 * (close - prev_close + 0.5*(close - open)
        //         + 0.25*(prev_close - prev_open)) / R * (K / limit_move)
        //   Simplified version (limit_move assumed large):
        let r_val = (high - prev.close).abs()
            .max((low - prev.close).abs())
            .max(high - low);
        let swing_index = if r_val > 0.0 {
            50.0 * ((close - prev.close)
                + 0.5 * (close - open)
                + 0.25 * (prev.close - prev.open))
                / r_val
        } else {
            0.0
        };

        // PL: AccumSwingIndex
        //   → manual: cumulative sum of SwingIndex
        self.accum_swing_index += swing_index;
        let accum_swing = self.accum_swing_index;

        // PL: Detrend(Close, 14)
        //   → close - SMA(close, 14)
        let detrend = close - self.detrend_sma14.next(close);

        // PL: PercentChange(Close, 14)
        //   → 100.0 * (close - close_14_ago) / close_14_ago
        let pct_change = if n > 14 && bars[n - 1 - 14].close != 0.0 {
            100.0 * (close - bars[n - 1 - 14].close) / bars[n - 1 - 14].close
        } else {
            0.0
        };

        // PL: UlcerIndex(Close, 14)
        //   → manual: sqrt of mean squared percent drawdown from rolling max
        let ulcer = {
            let lookback = n.min(14);
            let start = n - lookback;
            let mut max_close = f64::NEG_INFINITY;
            let mut sq_dd_sum = 0.0;
            for i in start..n {
                if bars[i].close > max_close { max_close = bars[i].close; }
                let pct_dd = if max_close > 0.0 {
                    100.0 * (bars[i].close - max_close) / max_close
                } else { 0.0 };
                sq_dd_sum += pct_dd * pct_dd;
            }
            (sq_dd_sum / lookback as f64).sqrt()
        };


        // =================================================================
        // SECTION 5 — Multi-Output Functions (2 functions)
        // =================================================================

        // PL: ParabolicSAR(0.02, 0.2, ...)
        //   → manual: full SAR with reference outputs for direction, af, ep
        //   Uses the same SAR state machine from Section 3 above.
        //   PL returns SAR value and populates ref vars (oSARDir, oAF, oEP).
        let parabolic_sar_val = self.sar;
        let parabolic_sar_dir = if self.sar_rising { 1.0 } else { -1.0 };
        let parabolic_sar_af = self.sar_af;
        let parabolic_sar_ep = self.sar_ep;

        // PL: LinearReg(Close, 20, 0, ...)
        //   → manual: least squares regression returning value, slope, r-squared
        //   PL populates ref vars (oLRValue, oLRSlope, oLRAngle, oRSquared).
        let lr_n = n.min(20);
        let lr_start = n - lr_n;
        let lr_len = lr_n as f64;
        let sum_x: f64 = (0..lr_n).map(|i| i as f64).sum();
        let sum_y: f64 = bars[lr_start..].iter().map(|b| b.close).sum();
        let sum_xy: f64 = bars[lr_start..].iter().enumerate()
            .map(|(i, b)| i as f64 * b.close).sum();
        let sum_xx: f64 = (0..lr_n).map(|i| (i as f64).powi(2)).sum();
        let sum_yy: f64 = bars[lr_start..].iter()
            .map(|b| b.close * b.close).sum();
        let denom = lr_len * sum_xx - sum_x * sum_x;
        let lr_slope = if denom != 0.0 {
            (lr_len * sum_xy - sum_x * sum_y) / denom
        } else { 0.0 };
        let lr_intercept = (sum_y - lr_slope * sum_x) / lr_len;
        let lr_value = lr_intercept + lr_slope * (lr_len - 1.0);
        let ss_tot = sum_yy - sum_y * sum_y / lr_len;
        let ss_res = sum_yy - lr_slope * sum_xy - lr_intercept * sum_y
            + lr_intercept * lr_slope * sum_x; // simplified
        let lr_r_squared = if ss_tot > 0.0 {
            1.0 - (ss_res / ss_tot).max(0.0)
        } else { 0.0 };


        // =================================================================
        // SECTION 6 — Volatility (3 functions)
        // =================================================================

        // PL: TrueRangeCustom(H, L, C)
        //   → manual: max(H-L, abs(H-C_prev), abs(L-C_prev))
        //   Same as TrueRange but accepts custom H/L/C inputs.
        let tr_custom = (high - low)
            .max((high - prev.close).abs())
            .max((low - prev.close).abs());

        // PL: VolatilityStdDev(20)
        //   → manual: stdev of log returns * sqrt(252)
        //   Annualized historical volatility.
        let vol_std_dev = {
            let lookback = n.min(20);
            if lookback >= 2 {
                let start = n - lookback;
                let log_returns: Vec<f64> = (start + 1..n)
                    .map(|i| (bars[i].close / bars[i - 1].close).ln())
                    .collect();
                let mean_lr: f64 = log_returns.iter().sum::<f64>()
                    / log_returns.len() as f64;
                let var: f64 = log_returns.iter()
                    .map(|r| (r - mean_lr).powi(2))
                    .sum::<f64>() / log_returns.len() as f64;
                var.sqrt() * (252.0_f64).sqrt()
            } else {
                0.0
            }
        };

        // PL: StandardDevAnnual(Close, 20, 1)
        //   → manual: stdev of close prices * sqrt(252)
        //   The "1" parameter selects population stdev (N divisor).
        let std_dev_annual = {
            let lookback = n.min(20);
            let start = n - lookback;
            let mean: f64 = bars[start..].iter().map(|b| b.close).sum::<f64>()
                / lookback as f64;
            let var: f64 = bars[start..].iter()
                .map(|b| (b.close - mean).powi(2))
                .sum::<f64>() / lookback as f64;
            var.sqrt() * (252.0_f64).sqrt()
        };


        // =================================================================
        // SECTION 7 — Price Extremes (2 functions)
        // =================================================================

        // PL: HighestFC(High, 20)
        //   → ta::indicators::Maximum::new(20)  *** ta-rs built-in ***
        let highest_fc = self.highest_fc.next(high);

        // PL: LowestFC(Low, 20)
        //   → ta::indicators::Minimum::new(20)  *** ta-rs built-in ***
        let lowest_fc = self.lowest_fc.next(low);


        // =================================================================
        // SECTION 8 — Pivot Detection (4 functions)
        // =================================================================

        // PL: PivotHighVS(1, High, 3, 3, 20)
        //   → manual: compare center bar's high with L left and R right neighbors
        //   Returns the pivot high value, or 0 if no pivot found in lookback.
        //   Params: (instance=1, price=High, left_strength=3, right_strength=3, lookback=20)
        let pivot_high_vs = {
            let left = 3_usize;
            let right = 3_usize;
            let scan_len = n.min(20);
            let mut result = 0.0_f64;
            if n > left + right {
                let start = if n > scan_len { n - scan_len } else { 0 };
                for i in ((start + left)..(n - right)).rev() {
                    let pivot = bars[i].high;
                    let is_pivot = (1..=left).all(|s| bars[i - s].high < pivot)
                        && (1..=right).all(|s| bars[i + s].high < pivot);
                    if is_pivot { result = pivot; break; }
                }
            }
            result
        };

        // PL: PivotLowVS(1, Low, 3, 3, 20)
        //   → manual: compare center bar's low with L left and R right neighbors
        let pivot_low_vs = {
            let left = 3_usize;
            let right = 3_usize;
            let scan_len = n.min(20);
            let mut result = 0.0_f64;
            if n > left + right {
                let start = if n > scan_len { n - scan_len } else { 0 };
                for i in ((start + left)..(n - right)).rev() {
                    let pivot = bars[i].low;
                    let is_pivot = (1..=left).all(|s| bars[i - s].low > pivot)
                        && (1..=right).all(|s| bars[i + s].low > pivot);

            else if dt_diff < 3600 { 252.0 * 6.5 * (3600.0 / dt_diff as f64) }
            else if dt_diff < 86400 { 252.0 * (86400.0 / dt_diff as f64) / 6.5 }
            else if dt_diff < 604800 { 252.0 }
            else if dt_diff < 2592000 { 52.0 }
            else { 12.0 }
        } else { 252.0 };

        // PL: LastBarOnChart
        //   → bar_index == data.len() - 1
        //   self.total_bars must be set by the execution engine before the run.
        let last_bar_on_chart = if self.total_bars > 0 {
            self.bar_count == self.total_bars - 1
        } else {
            false
        };


        // =================================================================
        // SECTION 17 — Custom Functions (10 functions)
        // =================================================================

        let bar_index = self.bar_count;

        // PL: StochRSI(Close, 14, 14) → manual: stochastic of RSI
        // Requires RSI values stored in a window; simplified version
        let rsi_val = /* use previously computed RSI */ 50.0_f64;
        let rsi_high = rsi_val; // placeholder — would track highest RSI over 14 bars
        let rsi_low = rsi_val;
        let stoch_rsi = if rsi_high != rsi_low { (rsi_val - rsi_low) / (rsi_high - rsi_low) } else { 0.0 };

        // PL: supertrend(10, 3) → manual: ATR-based trend bands
        let atr_val = self.atr_sum / 10.0_f64.max(1.0);
        let mid = (high + low) / 2.0;
        let upper_basic = mid + 3.0 * atr_val;
        let lower_basic = mid - 3.0 * atr_val;
        let st_val = if self.st_dir > 0 { lower_basic } else { upper_basic };

        if bar_index > 0 && volume < self.prev_volume && self.prev_close != 0.0 {
            self.nvi *= 1.0 + (close - self.prev_close) / self.prev_close;
        }

        // PL: PVI(1000) → manual: update only when volume rises
        if bar_index > 0 && volume > self.prev_volume && self.prev_close != 0.0 {
            self.pvi *= 1.0 + (close - self.prev_close) / self.prev_close;
        }

        // PL: Coppo(14, 11, 10) → manual: WMA of two ROC sums
        let roc14 = if self.prev_closes.len() >= 14 { (close - self.prev_closes[self.prev_closes.len()-14]) / self.prev_closes[self.prev_closes.len()-14] * 100.0 } else { 0.0 };
        let roc11 = if self.prev_closes.len() >= 11 { (close - self.prev_closes[self.prev_closes.len()-11]) / self.prev_closes[self.prev_closes.len()-11] * 100.0 } else { 0.0 };
        let coppo_raw = roc14 + roc11;
        let coppo_val = coppo_raw; // simplified — full WMA would smooth over 10 bars

        // PL: LWTI(Close, 10, 14) → manual
        let price_diff = if bar_index >= 10 { close - self.prev_closes[self.prev_closes.len().saturating_sub(10)] } else { 0.0 };
        let avg_range = (high - low).max(0.0001);
        let lwti_val = (price_diff / avg_range) * 50.0 + 50.0;

        // PL: TVI(Close, Volume, MinMove) → manual: cumulative directional volume
        let tick = 0.01_f64;
        if close >= self.prev_close + tick { self.tvi_dir = 1.0; }
        else if close <= self.prev_close - tick { self.tvi_dir = -1.0; }
        self.tvi += self.tvi_dir * volume;

        // PL: SharpeRatio → manual approximate: avg_return / std_return
        let daily_ret = if self.prev_close != 0.0 { (close - self.prev_close) / self.prev_close } else { 0.0 };
        let sharpe_approx = daily_ret; // simplified — full version needs rolling stats

        // PL: WRSI(14, Close) → manual Wilder RSI (same formula as standard RSI)
        // Uses Wilder smoothing: avg = prev*(N-1)/N + val/N
        let wrsi_val = 50.0_f64; // placeholder — same calc as RSI with Wilder smoothing

        // PL: NewMA(Close, 14) → manual: Heikin-Ashi + TEMA hybrid
        let ha_close_raw = (close + close + high + low) / 4.0;
        self.ha_open = (self.ha_open + ha_close_raw) / 2.0;
        let ha_max = high.max(self.ha_open);
        let ha_min = low.min(self.ha_open);
        let ha_c = (ha_close_raw + self.ha_open + ha_max + ha_min) / 4.0;
        let newma_val = ha_c; // simplified — full version chains triple EMA

        // Store close for ROC lookback used by Coppo and LWTI
        self.prev_closes.push(close);

        // STRATEGY LOGIC — demonstration using the computed values
        // =================================================================

        // Entry: combine several new indicators for a trend + momentum filter
        if self.position == 0 {

                && trix_val > 0.0
                && self.adx_val > 20.0

                    order_type: OrderType::Market,
