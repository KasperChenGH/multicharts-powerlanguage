// --- Swing Index (Wilder) ---
// PL: SwingIndex → manual Wilder formula
// Inputs: O, H, L, C of current and previous bar, limit move T
swingIndex(float o1, float h1, float l1, float c1, float o0, float h0, float l0, float c0, float T) =>
    k  = math.max(h0 - c1, l0 - c1)
    r1 = math.abs(h0 - c1) - 0.5 * math.abs(l0 - c1) + 0.25 * math.abs(c1 - o1)
    r2 = math.abs(l0 - c1) - 0.5 * math.abs(h0 - c1) + 0.25 * math.abs(c1 - o1)
    r3 = math.abs(h0 - l0) + 0.25 * math.abs(c1 - o1)
    r  = r1
    if math.abs(l0 - c1) > math.abs(h0 - c1) and math.abs(l0 - c1) > math.abs(h0 - l0)
        r := r2
    else if math.abs(h0 - l0) >= math.abs(h0 - c1) and math.abs(h0 - l0) >= math.abs(l0 - c1)
        r := r3
    r == 0 ? 0.0 : 50.0 * ((c0 - c1) + 0.5 * (c0 - o0) + 0.25 * (c1 - o1)) / r * (k / T)

// --- Fisher Transform ---
// PL: Fisher(Close) → 0.5 * ln((1+x)/(1-x)) where x is normalized price
fisherTransform(float src, int len) =>
    h = ta.highest(src, len)
    l = ta.lowest(src, len)
    raw = h == l ? 0.0 : 2.0 * ((src - l) / (h - l) - 0.5)
    x = math.max(-0.999, math.min(0.999, raw))
    0.5 * math.log((1.0 + x) / (1.0 - x))

// --- Fisher Inverse ---
// PL: FisherINV(x) → (exp(2x)-1)/(exp(2x)+1)
fisherInverse(float x) =>
    e2x = math.exp(2.0 * x)
    (e2x - 1.0) / (e2x + 1.0)

// --- Average Deviation ---
// PL: AvgDeviation(Close, 20) → mean absolute deviation
avgDeviation(float src, int len) =>
    m = ta.sma(src, len)
    sum = 0.0
    for i = 0 to len - 1
        sum += math.abs(src[i] - m)
    sum / len

// --- Kurtosis ---
// PL: Kurtosis(Close, 20) → excess kurtosis (4th central moment / variance^2 - 3)
kurtosis(float src, int len) =>
    m = ta.sma(src, len)
    s2 = 0.0
    s4 = 0.0
    for i = 0 to len - 1
        d = src[i] - m
        s2 += d * d
        s4 += d * d * d * d
    v = s2 / len
    v == 0 ? 0.0 : (s4 / len) / (v * v) - 3.0

// --- Skew ---
// PL: Skew(Close, 20) → 3rd central moment / stdev^3
skew(float src, int len) =>
    m = ta.sma(src, len)
    s2 = 0.0
    s3 = 0.0
    for i = 0 to len - 1
        d = src[i] - m
        s2 += d * d
        s3 += d * d * d
    sd = math.sqrt(s2 / len)
    sd == 0 ? 0.0 : (s3 / len) / (sd * sd * sd)

// --- Covariance ---
// PL: Covariance(Close, Volume, 20) → sample covariance
covariance(float x, float y, int len) =>
    mx = ta.sma(x, len)
    my = ta.sma(y, len)
    s = 0.0
    for i = 0 to len - 1
        s += (x[i] - mx) * (y[i] - my)
    s / len

// --- Trimmed Mean ---
// PL: TrimMean(Close, 20, 10) → mean after removing top/bottom 10%
trimMean(float src, int len, float pct) =>
    a = array.new_float(len)
    for i = 0 to len - 1
        array.set(a, i, src[i])
    array.sort(a, order.ascending)
    trim = math.round(len * pct / 100.0)
    sum = 0.0
    cnt = 0
    for i = trim to len - 1 - trim
        sum += array.get(a, i)
        cnt += 1
    cnt == 0 ? ta.sma(src, len) : sum / cnt

// --- Mode (approximate) ---
// PL: Mode(Close, 20, 1) → most frequent value (binned for continuous data)
modeApprox(float src, int len) =>
    // For continuous data, return the value closest to the median as approximation
    ta.percentile_nearest_rank(src, len, 50)

// --- Harmonic Mean ---
// PL: HarmonicMean(Close, 20) → N / sum(1/xi)
harmonicMean(float src, int len) =>
    s = 0.0
    for i = 0 to len - 1
        s += src[i] == 0 ? 0.0 : 1.0 / src[i]
    s == 0 ? 0.0 : len / s

// --- Ulcer Index ---
// PL: UlcerIndex(Close, 14) → sqrt(mean of squared pct drawdown)
ulcerIndex(float src, int len) =>
    hh = ta.highest(src, len)
    s = 0.0
    for i = 0 to len - 1
        pctDD = hh == 0 ? 0.0 : 100.0 * (src[i] - hh) / hh
        s += pctDD * pctDD
    math.sqrt(s / len)

// --- Mass Index ---
// PL: MassIndex(9, 25) → sum of EMA(range) / EMA(EMA(range)) over 25 bars
massIndex(int emaLen, int sumLen) =>
    rng     = high - low
    ema1    = ta.ema(rng, emaLen)
    ema2    = ta.ema(ema1, emaLen)
    ratio   = ema2 == 0 ? 1.0 : ema1 / ema2
    math.sum(ratio, sumLen)

// --- LRO (Least Recent Occurrence) ---
// PL: LRO(cond, lookback, nth) → bar offset of nth OLDEST true condition in lookback
lro(bool cond, int lookback, int nth) =>
    count = 0
    result = -1
    for i = lookback - 1 to 0
        if cond[i]
            count += 1
            if count == nth
                result := i
                break
    result

// --- MRO (Most Recent Occurrence) ---
// PL: MRO(cond, lookback, nth) → bars ago of nth most-recent true in lookback
mro(bool cond, int lookback, int nth) =>
    count = 0
    result = -1
    for i = 0 to lookback - 1
        if cond[i]
            count += 1
            if count == nth
                result := i
                break
    result

// --- SummationIf ---
// PL: SummationIf(cond, value, length) → sum of value where cond is true
summationIf(bool cond, float val, int len) =>
    s = 0.0
    for i = 0 to len - 1
        if cond[i]
            s += val[i]
    s

cDoji = math.abs(close - open) <= (high - low) * 0.05

// PL: C_Hammer_HangingMan(10, 33, oH, oHM) → manual: body/shadow ratios
// Hammer/Hanging Man: small body at top, long lower shadow
bodySize    = math.abs(close - open)
upperShadow = high - math.max(close, open)
lowerShadow = math.min(close, open) - low
totalRange  = high - low
cHammer     = totalRange > 0 and bodySize <= totalRange * 0.33 and lowerShadow >= totalRange * 0.60
cHangingMan = cHammer  // same shape; context (uptrend vs downtrend) determines name

// PL: C_BullEng_BearEng(10, oB, oBr) → manual: engulfing pattern
// Bullish: prior red candle, current green candle engulfs prior body
cBullEng = close[1] < open[1] and close > open and open <= close[1] and close >= open[1]
// Bearish: prior green candle, current red candle engulfs prior body
cBearEng = close[1] > open[1] and close < open and open >= close[1] and close <= open[1]

// PL: C_BullHar_BearHar(10, oB, oBr) → manual: harami pattern
// Bullish harami: large red candle followed by small green candle inside
cBullHar = close[1] < open[1] and close > open and open >= close[1] and close <= open[1]
// Bearish harami: large green candle followed by small red candle inside
cBearHar = close[1] > open[1] and close < open and open <= close[1] and close >= open[1]

// PL: C_MornDoji_EveDoji(10, 5, oM, oE) → manual: 3-bar doji star
// Morning doji star: red, doji gap-down, green gap-up
body1Red    = close[2] < open[2]
dojiMiddle  = math.abs(close[1] - open[1]) <= (high[1] - low[1]) * 0.05
body3Green  = close > open
cMornDoji   = body1Red and dojiMiddle and body3Green and close > (open[2] + close[2]) / 2.0
// Evening doji star: green, doji gap-up, red gap-down
body1Green  = close[2] > open[2]
body3Red    = close < open
cEveDoji    = body1Green and dojiMiddle and body3Red and close < (open[2] + close[2]) / 2.0

// PL: C_MornStar_EveStar(10, oM, oE) → manual: 3-bar star
// Morning star: red, small body gap-down, green gap-up
smallMiddle = math.abs(close[1] - open[1]) <= (high[1] - low[1]) * 0.33
cMornStar   = body1Red and smallMiddle and body3Green and close > (open[2] + close[2]) / 2.0
// Evening star: green, small body gap-up, red gap-down
cEveStar    = body1Green and smallMiddle and body3Red and close < (open[2] + close[2]) / 2.0

// PL: C_PierceLine_DkCloud(10, oP, oD) → manual: piercing/dark cloud
// Piercing line: red candle, green candle opens below prior low, closes above midpoint
cPierce = close[1] < open[1] and close > open and open < low[1] and close > (open[1] + close[1]) / 2.0
// Dark cloud cover: green candle, red candle opens above prior high, closes below midpoint
cDkCloud = close[1] > open[1] and close < open and open > high[1] and close < (open[1] + close[1]) / 2.0

// PL: C_ShootingStar(10, 33) → manual: shooting star
// Small body at bottom, long upper shadow
cShootStar = totalRange > 0 and bodySize <= totalRange * 0.33 and upperShadow >= totalRange * 0.60

// PL: C_3WhSolds_3BlkCrows(10, 33, oW, oC) → manual: 3 soldiers/crows
// Three white soldiers: 3 consecutive green candles with higher closes
c3WhSolds = close[2] > open[2] and close[1] > open[1] and close > open and close > close[1] and close[1] > close[2]
// Three black crows: 3 consecutive red candles with lower closes
c3BlkCrows = close[2] < open[2] and close[1] < open[1] and close < open and close < close[1] and close[1] < close[2]

// PL: StochRSI(Close, 14, 14) → compute RSI first, then stochastic of that RSI
// Stochastic of RSI — returns value in 0-1 range
rsiForStochRSI = ta.rsi(close, 14)
stochRSI = (rsiForStochRSI - ta.lowest(rsiForStochRSI, 14)) / (ta.highest(rsiForStochRSI, 14) - ta.lowest(rsiForStochRSI, 14))

// PL: supertrend(10, 3.0) → ta.supertrend(3.0, 10)
// Note: Pine order is (factor, atrPeriod), reversed from PL (ATRLen, Mult)
[stValue, stDir] = ta.supertrend(3.0, 10)

// PL: NVI(1000) → manual: cumulative index, updates only when volume decreases
// Negative Volume Index — adjusts by price change only on declining-volume bars
var float nviVal = 1000.0
if bar_index > 0 and volume < volume[1] and close[1] != 0
    nviVal := nviVal * (1.0 + (close - close[1]) / close[1])

// PL: PVI(1000) → manual: cumulative index, updates only when volume increases
// Positive Volume Index — adjusts by price change only on rising-volume bars
var float pviVal = 1000.0
if bar_index > 0 and volume > volume[1] and close[1] != 0
    pviVal := pviVal * (1.0 + (close - close[1]) / close[1])

// PL: Coppo(10, 14, 11) → ta.wma(ta.roc(close, 14) + ta.roc(close, 11), 10)
// Coppock Curve — WMA of sum of two rates of change
coppoVal = ta.wma(ta.roc(close, 14) + ta.roc(close, 11), 10)

// PL: LWTI(Close, 10, 14) → manual: Larry Williams Trading Index (0-100)
// Normalized average price change relative to average range
priceDiff = close - close[10]
avgDiff = ta.sma(priceDiff, 14)
avgRange = ta.sma(high - low, 14)
lwtiVal = avgRange != 0 ? (avgDiff / avgRange) * 50.0 + 50.0 : 50.0

// PL: TVI(Close, Volume, MinTickValue) → manual: cumulative directional volume
// Trade Volume Index — accumulates volume in direction of tick movement
var float tviVal = 0.0
var float tviDir = 0.0
tickSize = syminfo.mintick
if close >= close[1] + tickSize
    tviDir := 1.0
else if close <= close[1] - tickSize
    tviDir := -1.0
tviVal := tviVal + tviDir * volume

// PL: SharpeRatio → manual: approximate Sharpe from rolling returns
// Portfolio-level in PL; approximated here using bar-to-bar returns
retN = 20
dailyRet = close[1] != 0 ? (close - close[1]) / close[1] : 0.0
avgRet = ta.sma(dailyRet, retN)
retStd = ta.stdev(dailyRet, retN)
sharpeApprox = retStd != 0 ? avgRet / retStd : 0.0

// PL: WRSI(14, Close) → ta.rsi(close, 14)
// Wilder RSI — Pine's ta.rsi already uses Wilder smoothing (RMA)
wrsiVal = ta.rsi(close, 14)

// PL: NewMA(Close, 14) → manual: Heikin-Ashi TEMA hybrid
// Smoothed moving average using HA prices and double-TEMA
var float haO = close
haO := (nz(haO[1]) + (close + close + high + low) / 4.0) / 2.0
haMax = math.max(high, haO)
haMin = math.min(low, haO)
haC = ((close + close + high + low) / 4.0 + haO + haMax + haMin) / 4.0
ema1 = ta.ema(haC, 14)
tema1 = 3.0 * ema1 - 3.0 * ta.ema(ema1, 14) + ta.ema(ta.ema(ema1, 14), 14)
ema2 = ta.ema(tema1, 14)
tema2 = 3.0 * ema2 - 3.0 * ta.ema(ema2, 14) + ta.ema(ta.ema(ema2, 14), 14)
newMAVal = 2.0 * tema1 - tema2
