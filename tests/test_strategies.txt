If False Then Begin

    { ================================================================ }
    { === Strategy 1: MA Crossover === }
    { ================================================================ }
    fastMA = Average(Close, 10);
    slowMA = Average(Close, 20);

    If fastMA crosses over slowMA Then
        Buy ("S1_LE") 1 Contract Next Bar at Market;

    If fastMA crosses under slowMA Then
        Sell ("S1_LX") 1 Contract Next Bar at Market;

    { ================================================================ }
    { === Strategy 2: RSI + ATR Stop + Position Detection === }
    { ================================================================ }
    rsiVal = RSI(Close, 14);
    atrVal = AvgTrueRange(14);

    { Canonical position-transition pattern }
    If MarketPosition <> prevMP Then Begin
        Value1 = MarketPosition;
    End;
    prevMP = MarketPosition;

    If MarketPosition = 0 and rsiVal < 30 Then
        Buy ("S2_LE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then Begin
        stopPx = EntryPrice - 2 * atrVal;
        Sell ("S2_STP") 1 Contract Next Bar at stopPx Stop;
        If rsiVal > 70 Then
            Sell ("S2_TGT") 1 Contract Next Bar at Market;
    End;

    { ================================================================ }
    { === Strategy 3: Bollinger Band Breakout + Volume Filter === }
    { ================================================================ }
    upperBB = BollingerBand(Close, 20, 2);
    lowerBB = BollingerBand(Close, 20, -2);
    volAvg = Average(Volume, 20);

    If Close > upperBB and Volume > volAvg * 1.5 Then
        Buy ("S3_LE") 1 Contract Next Bar at Market;

    If Close < lowerBB and Volume > volAvg * 1.5 Then
        SellShort ("S3_SE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then
        Sell ("S3_LX") 1 Contract Next Bar at lowerBB Stop;

    If MarketPosition = -1 Then
        BuyToCover ("S3_SX") 1 Contract Next Bar at upperBB Stop;

    { ================================================================ }
    { === Strategy 4: Multi-Indicator + Risk Management === }
    { ================================================================ }
    adxVal = ADX(14);
    cciVal = CCI(20);

    If adxVal > 25 and cciVal crosses over -100 Then
        Buy ("S4_LE") 1 Contract Next Bar at Market;

    If adxVal > 25 and cciVal crosses under 100 Then
        SellShort ("S4_SE") 1 Contract Next Bar at Market;

    SetStopLoss(500);
    SetProfitTarget(1000);
    SetBreakEven(200);

    { ================================================================ }
    { === Strategy 5: Multi-Data Regime Filter === }
    { ================================================================ }
    { data1 = trading instrument, data2 = regime filter }
    If Close of data2 > Average(Close of data2, 50) Then Begin
        If RSI(Close, 14) < 30 Then
            Buy ("S5_LE") 1 Contract Next Bar at Market;
    End;

    If MarketPosition = 1 and RSI(Close, 14) > 70 Then
        Sell ("S5_LX") 1 Contract Next Bar at Market;

    { ================================================================ }
    { === Strategy 6: EMA Momentum + Limit Entry + Time Exit === }
    { ================================================================ }
    emaFast = XAverage(Close, 12);
    emaSlow = XAverage(Close, 26);
    momVal = Momentum(Close, 10);

    Once Begin
        Value1 = Close;
    End;

    If MarketPosition = 0 and momVal > 0 and emaFast > emaSlow Then
        Buy ("S6_LE") 1 Contract Next Bar at Close Limit;

    If MarketPosition = 1 and BarsSinceEntry >= 10 Then
        Sell ("S6_TX") 1 Contract Next Bar at Market;

    If MarketPosition = 1 and emaFast crosses under emaSlow Then
        Sell ("S6_LX") 1 Contract Next Bar at Market;

    { ================================================================ }
    { === Strategy 7: Donchian Channel Breakout === }
    { ================================================================ }
    hiChan = Highest(High, 20);
    loChan = Lowest(Low, 20);

    If Close > hiChan[1] Then
        Buy ("S7_LE") 1 Contract Next Bar at Market;

    If Close < loChan[1] Then
        SellShort ("S7_SE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then
        Sell ("S7_LX") 1 Contract Next Bar at loChan Stop;

    If MarketPosition = -1 Then
        BuyToCover ("S7_SX") 1 Contract Next Bar at hiChan Stop;

    { ================================================================ }
    { === Strategy 8: MACD Crossover + Trailing Stop === }
    { ================================================================ }
    macdLine = MACD(Close, 12, 26);
    signalLine = XAverage(MACD(Close, 12, 26), 9);

    If MarketPosition = 0 and macdLine crosses over signalLine Then
        Buy ("S8_LE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then Begin
        If BarsSinceEntry = 0 Then
            trailStop = EntryPrice - AvgTrueRange(14) * 3;

        If Close - AvgTrueRange(14) * 3 > trailStop Then
            trailStop = Close - AvgTrueRange(14) * 3;

        Sell ("S8_TS") 1 Contract Next Bar at trailStop Stop;

        If macdLine crosses under signalLine Then
            Sell ("S8_MX") 1 Contract Next Bar at Market;
    End;

    { ================================================================ }
    { === Strategy 9: Stochastic + Switch + Array + For + Contracts === }
    { ================================================================ }
    Value1 = Stochastic(High, Low, Close, 14, 3, 3, 1, fastK, fastD, slowK, slowD);

    { Shift array history using For loop }
    For Value2 = 4 DownTo 1 Begin
        kHistory[Value2] = kHistory[Value2 - 1];
    End;
    kHistory[0] = slowK;

    { Regime-based logic using Switch on MarketPosition }
    Switch (MarketPosition)
    Begin
        Case 0: Begin
            If slowK < 20 and slowD < 20 Then
                Buy ("S9_LE") 1 Contract Next Bar at Market;
            If slowK > 80 and slowD > 80 Then
                SellShort ("S9_SE") 1 Contract Next Bar at Market;
        End;
        Case 1:
            If slowK > 80 or CurrentContracts > 2 Then
                Sell ("S9_LX") 1 Contract Next Bar at Market;
        Case -1:
            If slowK < 20 or CurrentContracts > 2 Then
                BuyToCover ("S9_SX") 1 Contract Next Bar at Market;
    End;

    { ================================================================ }
    { === Strategy 10: While Loop + Date/Time Filter + Print/Alert === }
    { ================================================================ }
    { Count consecutive up bars using While loop }
    upCount = 0;
    Value3 = 1;
    While Value3 <= 5 and Close[Value3 - 1] > Close[Value3] Begin
        upCount = upCount + 1;
        Value3 = Value3 + 1;
    End;

    { Date/Time filter: trade only during session hours }
    If Time >= 930 and Time <= 1500 Then Begin
        If upCount >= 3 and MarketPosition = 0 Then
            Buy ("S10_LE") 1 Contract Next Bar at Market;
    End;

    If MarketPosition = 1 and Time > 1545 Then
        Sell ("S10_TX") 1 Contract Next Bar at Market;

    Print("upCount=", upCount, " MP=", MarketPosition);

        Alert("session close approaching");

    { ================================================================ }
    { === Strategy 11: DMI + Keltner Channel + Parabolic SAR === }
    { ================================================================ }
    dmiP = DMIPlus(14);
    dmiM = DMIMinus(14);

    { DirMovement multi-output: H, L, C, Len, 6 ref output vars }
    Value1 = DirMovement(High, Low, Close, 14,
        dmDmiP, dmDmiM, dmAdx, dmDIp, dmDIm, dmAdxr);

    keltUp = KeltnerChannel(Close, 20, 1.5);
    keltDn = KeltnerChannel(Close, 20, -1.5);
    sarVal = Parabolic(0.02);

    If dmiP > dmiM and dmAdx > 25 and Close > keltUp Then
        Buy ("S11_LE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then Begin
        Sell ("S11_SAR") 1 Contract Next Bar at sarVal Stop;
        If dmiM > dmiP Then
            Sell ("S11_DX") 1 Contract Next Bar at Market;
    End;

    If dmiM > dmiP and dmAdx > 25 and Close < keltDn Then
        SellShort ("S11_SE") 1 Contract Next Bar at Market;

    If MarketPosition = -1 Then
        BuyToCover ("S11_SC") 1 Contract Next Bar at sarVal Stop;

    { ================================================================ }
    { === Strategy 12: Williams %R + ROC + Volatility + StdDev === }
    { ================================================================ }
    wrVal = PercentR(14);
    rocVal = RateOfChange(Close, 10);
    volVal = Volatility(20);
    stdDevVal = StandardDev(Close, 20, 1);
    trVal = TrueRange;
    rng = Range;

    If wrVal < -80 and rocVal > 0 and volVal < stdDevVal * 2 Then

    If MarketPosition = 1 Then Begin
        If wrVal > -20 Then
            Sell ("S12_LX") 1 Contract Next Bar at Market;
        If trVal > rng * 3 Then
            Sell ("S12_VX") 1 Contract Next Bar at Market;
    End;

    { ================================================================ }
    { === Strategy 13: MFI + AccumDist + LinReg + WAverage === }
    { ================================================================ }
    mfiVal = MoneyFlow(14);
    adVal = AccumDist(14);
    linRegVal = LinearRegValue(Close, 20, 0);
    linRegSlp = LinearRegSlope(Close, 20);
    sumVal = Summation(Close, 10);
    cumVal = Cum(Volume);
    wavgVal = WAverage(Close, 20);
    midPtVal = MidPoint(Close, 20);

    If mfiVal < 20 and linRegSlp > 0 and Close > wavgVal Then
        Buy ("S13_LE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then Begin
        If mfiVal > 80 Then
            Sell ("S13_LX") 1 Contract Next Bar at Market;
        If Close < midPtVal Then
            Sell ("S13_MX") 1 Contract Next Bar at Market;
    End;

    { ================================================================ }
    { === Strategy 14: Swing + HighestBar + CountIF + Prices === }
    { ================================================================ }
    swHi = SwingHigh(1, High, 3, 10);
    swLo = SwingLow(1, Low, 3, 10);
    hiBar = HighestBar(High, 20);
    loBar = LowestBar(Low, 20);
    upBarsCount = CountIF(Close > Close[1], 10);
    avgPx = AvgPrice;
    typPx = TypicalPrice;

    If swLo > 0 and upBarsCount >= 6 and Close > typPx Then
        Buy ("S14_LE") 1 Contract Next Bar at Market;

    If MarketPosition = 1 Then Begin
        If swHi > 0 and hiBar > 15 Then
            Sell ("S14_LX") 1 Contract Next Bar at Market;
    End;

End;
